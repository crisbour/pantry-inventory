\usepackage{tikz}
\newcommand{\inventoryrow}[4]{%
  % #1: Product Name
  % #2: Supplier
  % #3: Number of items
  % #4: Unit per item
  %\newcommand{\boxwidth}{1.85} % Define  % #3: Number of Filled Boxes
  \pgfmathsetmacro{\boxwidth}{5.8 / #3} % Dynamically calculate box width
  \pgfmathtruncatemacro{\lastindex}{#3-1} % Calculate the upper limit of the loop
  \begin{tikzpicture}[baseline]
    % Product Name
    \node[anchor=west] (prod) at (0,0) {#1};
    % Supplier
    \node[anchor=west] (supp) at (8,0) {#2};
    % Boxes (start at x=8)
    \foreach \i in {0,...,\lastindex} {
      \draw[green,fill=none] (\boxoffset + \boxwidth*\i, -0.3) rectangle
        ++(0.2,0.6);
      \node[rotate=90, anchor=center, font=\scriptsize] at (\boxoffset +
        \boxwidth*\i + 0.1, 0) {open};
      \draw[blue,fill=none] (\boxoffset + 0.3 + \boxwidth*\i, -0.3) rectangle
        ++(\boxwidth-0.5, 0.6);
      \node[anchor=west] at (\boxoffset + \boxwidth*\i+0.25, 0) {#4};
    }
  \end{tikzpicture}%
}
